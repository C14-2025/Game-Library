pipeline {
    agent any

    tools {
        jdk 'JDK21'
        maven 'maven-3.9'
    }

    stages {

        stage('Checkout') {
            steps {
                echo "Clonando reposit√≥rio..."
                git branch: 'main', url: 'https://github.com/C14-2025/Game-Library.git'
            }
        }

        stage('Build') {
            steps {
                echo "Executando a build do projeto..."
                dir('Game-Library') {
                    bat "mvn clean package -DskipTests"
                }
            }
        }

        stage('Tests') {
            steps {
                echo "Executando os testes..."
                dir('Game-Library') {
                    bat "mvn test"
                }
            }
            post {
                always {
                    junit '**/target/surefire-reports/*.xml'
                }
            }
        }
    }

    post {
        success {
            echo "Pipeline finalizada com sucesso!"
        }
        failure {
            echo "Pipeline falhou!"
        }
    }
}
